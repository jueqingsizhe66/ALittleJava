42         public void init()
43         {
44             try
45             {
46                 InputStream is = Restaurant.class.getResourceAsStream("food.properties");
47                 Properties prop = new Properties();
48                 /**
49                  *  要知道 food.properties 的key-value的写法
50                  *  Prop对象就是一个map对象
51                  *  先前只写上一个值是错误的！ 必须写上

54                 //必须加在src底下
55                 //prop.load(new FileInputStream("food.properties"));
56                 try
57                 {
58                     prop.load(is);
59                 } catch (IOException e)
60                 {
61                     // TODO Auto-generated catch block
62                     e.printStackTrace();
63                 }
64                 Collection  coll1 = prop.values();
65                 for(Object foodClassName:coll1)
66                 {
67                     Class temp;
68     
69                     temp = Class.forName((String)(foodClassName));
70     
71                     Food food = (Food)temp.newInstance();
72                     foods.add(food);
73                 }
74             }catch (ClassNotFoundException e)
75             {
76                 // TODO Auto-generated catch block
77                 System.out.println("未找到对应的类");
78             }catch (InstantiationException e)
79             {
80                 // TODO Auto-generated catch block
81                 System.out.println("产生实例的异常");
82             } catch (IllegalAccessException e)
83             {
84                 // TODO Auto-generated catch block
85                 e.printStackTrace();
86             }
