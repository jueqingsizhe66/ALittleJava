132     public abstract class PieD { //比萨饼
133     	
134     	//定义两个访问者
135     	RemV remFn = new RemV();
136     	SubstV subFn = new SubstV();
137     	abstract PieD remA(Object o);
138     	abstract PieD substFish(Object n, Object o);
139     }
140     
141     class Bot extends PieD{
142     
143     	@Override
144     	PieD remA(Object o) {
145     		// TODO Auto-generated method stub
146     		return remFn.forBot(o);
147     	}
148     
149     	@Override
150     	PieD substFish(Object n, Object o) {
151     		// TODO Auto-generated method stub
152     		return subFn.forBot(n, o);
153     	}
154     	
155     }

157     class Top extends PieD{
158     
159     	Object t;
160     	PieD r;
161     	public Top(Object t,PieD r) {
162     		// TODO Auto-generated constructor stub
163     		this.t = t;
164     		this.r = r;
165     	}
166     	@Override
167     	PieD remA(Object f) {
168     		// TODO Auto-generated method stub
169     		return remFn.forTop(t, r, f);
170     	}
171     	@Override
172     	PieD substFish(Object n, Object o) {
173     		// TODO Auto-generated method stub
174     		return subFn.forTop(t, r, n, o);
175     	}
176     	
177     }
